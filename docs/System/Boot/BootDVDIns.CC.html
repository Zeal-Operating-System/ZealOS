#include "BootDVD"
#include "DiskISORedSea"

#help_index "Install"

#define KERNEL_BIN_C                "Kernel.BIN.C"
#define BOOT_DIR                    "/Boot"
#define BOOT_DIR_DVD_KERNEL_BIN_C   BOOT_DIR "/DVD" KERNEL_BIN_C

U0 MakeAll()
{
    if (Comp("/Compiler/Compiler", "Compiler",, ':'))
        throw;
    if (Comp("/Kernel/Kernel", "Kernel",, ':'))
        throw;
}

public U0 BootDVDIns(U8 drv_let=0)
{//See ::/Misc/DoDistro.CC.
    try
    {
        if (!Drive(drv_let))
            throw;
        MakeAll;
        Move("/Kernel/Kernel.BIN", BOOT_DIR_DVD_KERNEL_BIN_C);
    }
    catch
    {
        PutExcept;
        Beep;
    }
}
